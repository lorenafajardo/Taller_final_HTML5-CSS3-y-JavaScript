Se está creando una aplicación que va a estar conectada con un prototipo que permita almacenar contactos telefónicos en el dispositivo. Para ello cada contacto debe contener nombre completo, teléfono y organización. Se requiere que la aplicación permita añadir 3 contactos verificando que el número no esté almacenado, buscar contactos almacenados y eliminar contactos si el usuario lo requiere. Recuerde que el sistema debe terminar cuando el usuario así lo indique.

    FinSubProceso

                    si (contactoBuscar = nombrei) Entonces
                        busquedai <- informacioni(nombrei, telefonoi, organizacioni); 
                    FinSi
                    
                    si (contactoBuscar = nombreii) Entonces
                        busquedai <- informacioni(nombreii, telefonoii, organizacionii); 
                    FinSi
                    
                    si (contactoBuscar = nombreiii) Entonces
                        busquedai <- informacioni(nombreiii, telefonoiii, organizacioniii); 
                    FinSi
                SiNo
                    Escribir "El contacto no se encuenntra almacenado";
                FinSi
            3:
                Escribir "        Eliminar Contacto    ";
                
                contactoEliminar <- capturaiii() ;
                
                si ( (contactoEliminar = nombrei) o (contactoEliminar = nombreii) o (contactoEliminar = nombreiii)) Entonces
                    
                    si (contactoEliminar = nombrei) Entonces
                        nombrei<- "";
                        telefonoi<- ""; 
                        organizacioni <- "";
                    FinSi
                    
                    si (contactoEliminar = nombreii) Entonces
                        nombreii<- "";
                        telefonoii<- ""; 
                        organizacionii <- "";
                    FinSi
                    
                    si (contactoEliminar = nombreiii) Entonces
                        nombreiii<- "";
                        telefonoiii<- ""; 
                        organizacioniii <- ""; 
                    FinSi
                    Escribir  " El contacto se ha eliminado con exito";
                SiNo
                    Escribir "El contacto no se encuenntra almacenado";
                FinSi
                
        FinSegun
    
        Hasta Que opcion = 4  
        
        Escribir "                   Vuelva pronto";
        
    FinProceso
